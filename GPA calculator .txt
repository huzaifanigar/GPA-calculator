#include <iostream>
using namespace std;

float calculateGPA(float marks[], int n) {
    float sum = 0;
    for(int i=0; i<n; i++) {
        sum += marks[i];
    }
    return sum / n; // average marks = GPA
}

int main() {
    int subjects;
    cout << "Enter number of subjects: ";
    cin >> subjects;

    float marks[subjects];
    cout << "Enter marks for each subject: \n";
    for(int i=0; i<subjects; i++) {
        cin >> marks[i];
    }

    float gpa = calculateGPA(marks, subjects);
    cout << "Your GPA is: " << gpa << endl;
    return 0;
}
